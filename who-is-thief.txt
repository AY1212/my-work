#include<stdio.h>
int main()
{
    int i, A=1, B=0, C=0, D=0;  /*先假定甲是窃贼*/
    for(i=1; i<=4; i++)
        if(B+D==1 && B+C==1 && A+B==1)  /*测试甲乙丙丁谁是窃贼，符合该条件的即为窃贼*/
            break;
        else
        {
            if(i == 1)
            {
                A=0;  /*甲不是窃贼，测试乙是否是窃贼*/
                B=1;
            }
            if(i == 2)
            {
                B=0;  /*甲乙均不是窃贼，测试丙是否是窃贼*/
                C=1;

            }
            if(i == 3)
            {
                C=0;  /*甲乙丙都不是窃贼，测试丁是否是窃贼*/
                D=1;
            }
        }
    /*输出结果*/
    printf("判断结果：\n");
    if(i == 1)
        printf("甲是窃贼\n");
    if(i == 2)
        printf("乙是窃贼\n");
    if(i == 3)
        printf("丙是窃贼\n");
    if(i == 4)
        printf("丁是窃贼\n");
   
    return 0;
}
